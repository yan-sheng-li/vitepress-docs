ж•ҙеҗҲ WebSocket ж•ҷзЁӢз¬”и®° вҖӢ

WebSocket жҳҜдёҖз§ҚеңЁе®ўжҲ·з«Ҝе’ҢжңҚеҠЎз«Ҝд№Ӣй—ҙе»әз«ӢжҢҒд№…иҝһжҺҘзҡ„йҖҡдҝЎеҚҸи®®пјҢйқһеёёйҖӮеҗҲе®һж—¶иҒҠеӨ©гҖҒеңЁзәҝйҖҡзҹҘзӯүеә”з”ЁеңәжҷҜгҖӮдёӢйқўжҲ‘д»¬е°ҶйҖҡиҝҮдёҖдёӘз®ҖеҚ•зҡ„ Demo жқҘд»Ӣз»ҚеҰӮдҪ•еңЁ Spring Boot дёӯйӣҶжҲҗ WebSocketгҖӮ

рҹ§ұ дёҖгҖҒж·»еҠ дҫқиө– вҖӢ

еңЁ pom.xml дёӯж·»еҠ  WebSocket зҡ„ starterпјҡ

xml
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-websocket</artifactId>
</dependency>

вҡҷпёҸ дәҢгҖҒй…ҚзҪ® WebSocket вҖӢ

ж–°е»әдёҖдёӘй…ҚзҪ®зұ» WebSocketConfig.javaпјҡ

java
package com.base.config;

import org.springframework.context.annotation.Configuration;
import org.springframework.messaging.simp.config.MessageBrokerRegistry;
import org.springframework.web.socket.config.annotation.EnableWebSocketMessageBroker;
import org.springframework.web.socket.config.annotation.StompEndpointRegistry;
import org.springframework.web.socket.config.annotation.WebSocketMessageBrokerConfigurer;

@Configuration
@EnableWebSocketMessageBroker // еҗҜз”Ё WebSocket ж¶ҲжҒҜд»ЈзҗҶ
public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {

    @Override
    public void configureMessageBroker(MessageBrokerRegistry config) {
        // й…ҚзҪ®ж¶ҲжҒҜд»ЈзҗҶ
        config.enableSimpleBroker("/topic"); // з”ЁдәҺи®ўйҳ…ж¶ҲжҒҜзҡ„еүҚзјҖ
        config.setApplicationDestinationPrefixes("/app"); // е®ўжҲ·з«ҜеҸ‘йҖҒж¶ҲжҒҜзҡ„еүҚзјҖ
    }

    @Override
    public void registerStompEndpoints(StompEndpointRegistry registry) {
        // жіЁеҶҢ STOMP з«ҜзӮ№пјҢејҖеҗҜ SockJS ж”ҜжҢҒ
        registry.addEndpoint("/ws").setAllowedOrigins("*").withSockJS();
    }
}

иҜҙжҳҺпјҡ

  • /wsпјҡWebSocket зҡ„иҝһжҺҘз«ҜзӮ№
  • /topicпјҡе№ҝж’ӯи·Ҝеҫ„пјҲзұ»дјјиҒҠеӨ©е®Өпјү
  • /appпјҡе®ўжҲ·з«ҜеҸ‘йҖҒж¶ҲжҒҜж—¶дҪҝз”Ёзҡ„и·Ҝеҫ„еүҚзјҖ

рҹ§‘вҖҚрҹ’» дёүгҖҒеҲӣе»ә Controller жҺҘж”¶ж¶ҲжҒҜ вҖӢ

еҲӣе»ә WebSocketController.java еӨ„зҗҶеүҚз«ҜеҸ‘йҖҒзҡ„ж¶ҲжҒҜе№¶иҝ”еӣһпјҡ

java
package com.base.controller;

import org.springframework.messaging.handler.annotation.MessageMapping;
import org.springframework.messaging.handler.annotation.SendTo;
import org.springframework.stereotype.Controller;

@Controller
public class WebSocketController {

    @MessageMapping("/sendMessage") // еүҚз«ҜеҸ‘йҖҒеҲ° /app/sendMessage
    @SendTo("/topic/messages") // е№ҝж’ӯеҲ° /topic/messages
    public String handleMessage(String message) {
        System.out.println("Received message: " + message);
        return "Echo: " + message;
    }
}

рҹҢҗ еӣӣгҖҒеүҚз«ҜйЎөйқўпјҲHTML+JSпјү вҖӢ

еҲӣе»әдёҖдёӘз®ҖеҚ•зҡ„ HTML йЎөйқўпјҢз”ЁдәҺжөӢиҜ• WebSocket иҝһжҺҘпјҡ

html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>WebSocket жөӢиҜ•</title>
    <script src="https://cdn.jsdelivr.net/npm/sockjs-client@1/dist/sockjs.min.js"></script>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/stomp.js/2.3.3/stomp.min.js"></script>
</head>
<body>
    <div>
        <button onclick="connect()">иҝһжҺҘ WebSocket</button>
        <button onclick="disconnect()">ж–ӯејҖиҝһжҺҘ</button>
    </div>
    <div>
        <input type="text" id="messageInput" placeholder="иҫ“е…Ҙж¶ҲжҒҜ">
        <button onclick="sendMessage()">еҸ‘йҖҒж¶ҲжҒҜ</button>
    </div>
    <div id="messages"></div>

    <script>
        var stompClient = null;

        function connect() {
            var socket = new SockJS('/ws');
            stompClient = Stomp.over(socket);
            stompClient.connect({}, function(frame) {
                console.log('Connected: ' + frame);
                stompClient.subscribe('/topic/messages', function(response) {
                    showMessage('ж”¶еҲ°ж¶ҲжҒҜ: ' + response.body);
                });
            });
        }

        function disconnect() {
            if (stompClient !== null) {
                stompClient.disconnect();
            }
            console.log("е·Іж–ӯејҖиҝһжҺҘ");
        }

        function sendMessage() {
            var message = document.getElementById('messageInput').value;
            stompClient.send("/app/sendMessage", {}, message);
        }

        function showMessage(message) {
            var messageDiv = document.getElementById('messages');
            var p = document.createElement('p');
            p.textContent = message;
            messageDiv.appendChild(p);
        }
    </script>
</body>
</html>
